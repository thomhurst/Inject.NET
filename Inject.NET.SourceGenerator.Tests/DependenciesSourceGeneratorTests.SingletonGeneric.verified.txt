using System;
using System.Linq;
using Inject.NET.Enums;
using Inject.NET.Extensions;
using Inject.NET.Services;

namespace Inject.NET.SourceGenerator.Sample.ServiceProviders;

public class SingletonGenericServiceRegistrar : ServiceRegistrar
{
	public SingletonGenericServiceRegistrar()
	{
		Register(new global::Inject.NET.Models.ServiceDescriptor
		{
			ServiceType = typeof(global::Inject.NET.SourceGenerator.Sample.Models.Generic<>),
			ImplementationType = typeof(global::Inject.NET.SourceGenerator.Sample.Models.Generic<>),
			Lifetime = Inject.NET.Enums.Lifetime.Singleton,
			Factory = (scope, type, key) =>
			 Activator.CreateInstance(typeof(global::Inject.NET.SourceGenerator.Sample.Models.Generic<>).MakeGenericType(type.GenericTypeArguments), [ ..type.GenericTypeArguments.Select(x => scope.GetService(x)) ])
		});

	}
}

  
using System;
using System.Threading.Tasks;
using Inject.NET.Enums;
using Inject.NET.Interfaces;

namespace Inject.NET.SourceGenerator.Sample.ServiceProviders;

public partial class SingletonGeneric
{
	public static ValueTask<IServiceProviderRoot> BuildAsync() =>
		new SingletonGenericServiceRegistrar().BuildAsync();
}
